def gerar_grafico_barras(dados: pd.Series, titulo="", yLabel="", xLabel="", horizontal_orientation=False, labels=""):
    '''Fun√ß√£o para gerar gr√°fico de barras
    INPUT:
        dados = Dados para serem mostrados no gr√°fico. Obrigat√≥riamente devem ser do tipo pd.Series
        labels = Nome para ser utilizado como tick_label, fica abaixo de cada variavel. Caso n√£o seja informado, pega o label dos dados.
        titulo = T√≠tulo do gr√°fico.
        yLabel = Nome do label que fica a esquerda do gr√°fico.
        xLabel = Nome do label que fica abaixo do gr√°fico.
        horizontal_orientation = Indica se o gr√°fico deve ser gerado na Vertical(Padr√£o) ou Horizontal.
    OUTPUT:
        Gr√°fico de barras utilizando os dados informados.
    '''
    locations = np.arange(dados.index.to_series().count()) #Pega a quantidade de variaveis pela quantidade de index
    heights = dados #altura do gr√°fico s√£o os dados passados na chamada da fun√ß√£o
    if(not labels): #Se os labels n√£o foram informados
        labels = dados.index.to_series() #labels ser√£o os index dos dados
    plt.figure(figsize=(10,5)) #configura o tamanho do gr√°fico
    if(horizontal_orientation): #Verifica se o gr√°fico deve ser gerado na horizontal
        plt.barh(locations, heights, tick_label=labels)
    else: #gr√°fico gerado na vertical
        plt.bar(locations, heights, tick_label=labels)
    plt.title(titulo) #Define o t√≠tulo do gr√°fico 
    plt.ylabel(yLabel) #Define o texto que fica na esquerda do gr√°fico
    plt.xlabel(xLabel) #Define o texto que fica abaixo do gr√°fico
